    i:integer;

function mymod(x,y:integer):integer;
	begin
		mymod:=x-x/y*y;
	end;

	begin
		temp:=x;
		x:=y;
		y:=temp;
	end;

begin

	begin
		if i = 2 then
			begin
				x:=45;
				y:=27;
			end
		else
			begin
				x:=5;
				y:=0;
			end;
		a:=x;
		b:=y;
		if y <> 0 then
			begin
				do
					begin
						cc:=b;
						b:=mymod(a,b);

					end

			end;
